CREATE TABLE trains (
	train_id INT PRIMARY KEY AUTO_INCREMENT,
	train_start INT NOT NULL,
	train_end INT NOT NULL,
	train_direction tinyint(1),
	train_days tinyint(1)
	);

# train_days 0 is M-F, 1 is SSH
